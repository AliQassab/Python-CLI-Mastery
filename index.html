‚öôÔ∏è Mastering Argparse

Argparse is Python's built-in library for handling command-line arguments. Think of it as Python's version of Commander.js!

1 Create the Parser 
parser = argparse.ArgumentParser(
    prog="mycommand",           # Name of your command
    description="What it does"   # Shows in --help
)
2 Add Flags (Optional Arguments) 
# Boolean flag (True/False)
parser.add_argument("-v", "--verbose", action="store_true", help="Verbose output")

# Flag that takes a value
parser.add_argument("-n", "--number", type=int, help="A number")

# Flag with default value
parser.add_argument("-f", "--format", default="json", help="Output format")
3 Add Arguments (Required/Positional) 
# Required argument
parser.add_argument("filename", help="File to process")

# Optional argument with default
parser.add_argument("output", nargs="?", default="output.txt", help="Output file")

# Multiple files
parser.add_argument("files", nargs="+", help="One or more files")
4 Parse and Use 
args = parser.parse_args()

# Now use the arguments
if args.verbose:
    print("Verbose mode enabled!")

print(f"Processing {args.filename}")
for file in args.files:
    print(f"File: {file}")

üîç Real Example: Our WC Command File Processing

WC Command File Processing
def count_stats(text):
    # Count lines (split creates list, subtract 1)
    lines = len(text.split("\n")) - 1
    
    # Count words (split() handles multiple spaces)
    words = len(text.strip().split())
    
    # Count bytes in UTF-8 encoding
    bytes_count = len(text.encode("utf-8"))
    
    return {"lines": lines, "words": words, "bytes": bytes_count}

# Usage in main loop
for filename in args.files:
    try:
        with open(filename, "r", encoding="utf-8") as f:
            text = f.read()
        
        stats = count_stats(text)
        print(f"{stats['lines']:7} {stats['words']:7} {stats['bytes']:7} {filename}")
        
    except FileNotFoundError:
        print(f"wc: {filename}: No such file or directory", file=sys.stderr)

Every CLI tool follows similar patterns. Master these and you can build any command-line tool!

1 The Universal CLI Structure 
#!/usr/bin/env python3
import argparse
import sys

def main():
    # 1. Parse arguments
    parser = argparse.ArgumentParser(...)
    args = parser.parse_args()
    
    # 2. Process each input
    for item in args.inputs:
        try:
            # 3. Do the work
            result = process_item(item)
            # 4. Output result
            print(result)
        except Exception as e:
            # 5. Handle errors
            print(f"Error: {e}", file=sys.stderr)

if __name__ == "__main__":
    main()
2 File Processing Pattern 
# Pattern: Read ‚Üí Process ‚Üí Output
for filename in args.files:
    try:
        # READ
        with open(filename, "r", encoding="utf-8") as f:
            content = f.read()
        
        # PROCESS
        lines = content.split("\n")
        processed_lines = [process_line(line) for line in lines]
        
        # OUTPUT
        for line in processed_lines:
            print(line)
            
    except FileNotFoundError:
        print(f"Error: {filename} not found", file=sys.stderr)
3 Conditional Output Pattern 
# Pattern: Different output based on flags
def format_output(data, args):
    output = ""
    
    # Show all if no specific flags
    show_all = not (args.lines or args.words or args.bytes)
    
    if args.lines or show_all:
        output += f"{data['lines']:7}"
    if args.words or show_all:
        output += f"{data['words']:7}"
    if args.bytes or show_all:
        output += f"{data['bytes']:7}"
    
    return output + f" {filename}"
4 Accumulator Pattern (Totals) 
# Pattern: Keep running totals
total = {"lines": 0, "words": 0, "bytes": 0}

for filename in args.files:
    # Process file and get stats
    stats = process_file(filename)
    
    # Add to totals
    total["lines"] += stats["lines"]
    total["words"] += stats["words"]
    total["bytes"] += stats["bytes"]
    
    # Show individual file stats
    print(format_output(stats, filename))

# Show totals if multiple files
if len(args.files) > 1:
    print(format_output(total, "total"))

üéØ Complete Side-by-Side: Cat Command

JavaScript Version 

import { program } from "commander";
import { readFile } from "node:fs/promises";

program
  .name("cat")
  .option("-n", "Number lines")
  .argument("<files...>", "Files")
  .parse();

const options = program.opts();
const files = program.args;

for (const file of files) {
  try {
    const content = await readFile(file, "utf-8");
    const lines = content.split("\n");
    
    lines.forEach((line, i) => {
      let prefix = "";
      if (options.n) {
        prefix = (i + 1).toString().padStart(6) + "  ";
      }
      console.log(prefix + line);
    });
  } catch (error) {
    console.error(`Error: ${error.message}`);
  }
}

Python Version 

#!/usr/bin/env python3
import argparse
import sys

def main():
    parser = argparse.ArgumentParser(prog="cat")
    parser.add_argument("-n", action="store_true")
    parser.add_argument("files", nargs="+")
    args = parser.parse_args()
    
    for file in args.files:
        try:
            with open(file, "r", encoding="utf-8") as f:
                content = f.read()
            lines = content.split("\n")
            
            for i, line in enumerate(lines):
                prefix = ""
                if args.n:
                    prefix = str(i + 1).rjust(6) + "  "
                print(prefix + line)
        except Exception as error:
            print(f"Error: {error}", file=sys.stderr)

if __name__ == "__main__":
    main()
